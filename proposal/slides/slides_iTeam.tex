\begin{frame}
  \[ 
  R(\xx) = \sum_{t,\xx_s} G^+(\xx,\xx_s,t) G^-(\xx,\xx_s,t)
  \]
\sep  
  \itab{$G^+$: downgoing wavefield}\\
  \itab{$G^-$: upgoing wavefield}\\
\end{frame}




\begin{frame}
  \[ 
  R(\xx) = \sum_{t,\xx_s} \US(\xx,\xx_s,t) \UR(\xx,\xx_s,t)
  \]
\sep  
  \itab{$u_s$: source wavefield}\\
  \itab{$u_r$: receiver wavefield}
\end{frame}

\begin{frame}
  \[ 
  R(\xx,\hh,\tau) = \sum_{t} \US(\xx +\hh,t+\tau) \UR(\xx-\hh,t-\tau)
  \]
\sep  
  \itab{$\bf u_s$: source wavefield}\\
  \itab{$\bf u_r$: receiver wavefield}\\
  {\itab{$\hh,\tau$: space and time-lags}} 
\end{frame}



\begin{frame}
  \[
    {\bf r}_{\hh,\tau} = {\bf I_s} {\bf u_r}  
  \]
\sep  
  \itab{$\bf u_s$: source wavefield}\\
  \itab{$\bf u_r$: receiver wavefield}\\
  {\itab{$\hh,\tau$: space and time-lags}} 
\end{frame}


\begin{frame} \frametitle{Multi Dimensional Deconvolution}
  \[
   {\bf I_s}^\top {\bf r}_{\hh,\tau} \approx  {\bf u_r}  
  \]
\sep  
  \itab{$\bf u_s$: source wavefield}\\
  \itab{$\bf u_r$: receiver wavefield}\\
  {\itab{$\hh,\tau$: space and time-lags}} 
\end{frame}

\begin{frame} \frametitle{Multi Dimensional Deconvolution}
  \[
   {\bf r}_{\hh,\tau} = \lp {\bf I_s} {\bf I_s}^\top \rp^{-1} {\bf I_s}  {\bf u_r}  
  \]
\sep  
  \itab{$\bf u_s$: source wavefield}\\
  \itab{$\bf u_r$: receiver wavefield}\\
  {\itab{$\hh,\tau$: space and time-lags}} 
\end{frame}
